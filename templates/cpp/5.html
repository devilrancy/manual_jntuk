#include<iostream.h>
#include<stdlib.h>
void main()
{
  int ch1,ch2,inum1,inum2,ires;
  float fnum1,fnum2,fres;

  cout<<"\nSelect the 1st variable type\n";
  cout<<"1-int\n2-float\n3-char\nEnter your choice:";
  cin>>ch1;

  switch(ch1)
  {
    case 1:cout<<"\nEnter the Number:";
        cin>>inum1;
        break;
    case 2:cout<<"\nEnter the Number:";
        cin>>fnum1;
        break;

        exit(0);
        break;
    default:cout<<"\nPlease enter correct choice....Exiting....";
          exit(0);

  }

  cout<<"\nSelect the 2nd variable type\n";
  cout<<"1-int\n2-float\n3-char\nEnter your choice:";
  cin>>ch2;

  switch(ch2)
  {
    case 1:cout<<"\nEnter the Number:";
        cin>>inum2;
        if(ch1==1)
        {
          ires=inum1*inum2;
          cout<<"Result is:"<<ires;
        }
        else if(ch1==2)
        {
          fres=fnum1*inum2;
          cout<<"Result is:"<<fres;
        }
        break;
    case 2:cout<<"\nEnter the Number:";
        cin>>fnum2;
        if(ch1==1)
          fres=inum1*fnum2;

        break;

    default:cout<<"\nPlease enter correct choice....Exiting....";

}
