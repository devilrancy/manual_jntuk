#include<iostream.h>

class student
{
  private:
    int roll;
    char name[20];
    int age;
  public:
    student()
    {
      roll=0;
      age=0;
    }
    student(int r,int a)
    {
      roll=r;
      age=a;
    }

    void show()
    {
      cout<<"\nRoll:"<<roll;
      cout<<"\nAge:"<<age;
    }

};

void main()
{
  student obj;
  obj.show();
  student obj1(2,10);
  obj1.show();
}
