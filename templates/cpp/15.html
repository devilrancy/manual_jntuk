#include<iostream.h>

template<class T>

void swapVariables(T &var1, T &var2)
{
  T temp;           
  temp = var1;
  var1 = var2;
  var2 = temp;
}

int main( )
{
  float float1 = 3.5, float2 = 5.6;
  int int1=2,int2=10;
  char char1='a',char2='b';

 cout<<"\nBefore Swapping\n";
 cout<<"Float 1:"<<float1<<"\nFloat 2:"<<float2;
 swapVariables(float1, float2);
 cout<<"\nAfter Swapping\n";
 cout<<"Float 1:"<<float1<<"\nFloat 2:"<<float2;

 cout<<"\n\n\nBefore Swapping\n";
 cout<<"Int 1:"<<int1<<"\nInt 2:"<<int2;
 swapVariables(int1, int2);
 cout<<"\nAfter Swapping\n";
 cout<<"Int 1:"<<int1<<"\nInt 2:"<<int2;

 cout<<"\n\n\nBefore Swapping\n";
 cout<<"char 1:"<<char1<<"\nchar 2:"<<char2;
 swapVariables(char1, char2);
 cout<<"\nAfter Swapping\n";
 cout<<"char 1:"<<char1<<"\nchar 2:"<<char2;




}
