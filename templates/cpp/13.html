#include<iostream.h>
class stu{
  private:
    int roll;
    char name[20];
  public:
    void insert();
    friend void display(stu s);
}obj;
void stu::insert(){
  cout<<"\nEnter Roll No:";
  cin>>roll;

  cout<<"\nEnter Name:";
  cin>>name;

}
void display(stu s){
  cout<<"\nRoll:"<<s.roll;
  cout<<"\nName:"<<s.name;

}
void main(){
   obj.insert();
   display(obj);
}
