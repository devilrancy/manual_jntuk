#include<iostream.h>

class Inc
{
  int val1;
  int val2;
  int res;

public:
  Inc (int v1,int v2)
  {
    val1=v1;
    val2=v2;
    cout<<"\nConstructor with two parameter is working.....";
  }

  Inc (int r)
  {
    res=r;
    cout<<"\nConstructor with single parameter is working.....";
  }

  Inc operator+()
  {

    res=val1+val2;
    cout<<"\nConstructor operator overloading is working.....";
    return Inc(res);
  }



  void display()
  {
    cout<<"\nValue:"<<res;
  }

};
void main()
{
  Inc obj(10,12);
