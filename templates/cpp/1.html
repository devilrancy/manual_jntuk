#include<iostream.h>
#include<conio.h>

int num=10;

void main()
{
  cout<<"Number is having Global Scope:"<<num;

  {
    int num=15;
    cout<<"\nNumber is having Local Scope:"<<num;
  }

  cout<<"\nNumber again having Local Scope:"<<num;
}
