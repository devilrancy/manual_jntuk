#include<iostream.h>
void swap(int &num1,int &num2,char &ch1,char &ch2,float &f1,float &f2);

void main()
{
  int num1,num2;
  char ch1,ch2;
  float f1,f2;

  cout<<"\nEnter 1st number:";
  cin>>num1;

  cout<<"\nEnter 2nd number:";
  cin>>num2;

  cout<<"\nEnter 1st character:";
  cin>>ch1;

  cout<<"\nEnter 2nd character:";
  cin>>ch2;

  cout<<"\nEnter 1st floating number:";
  cin>>f1;

  cout<<"\nEnter 2nd floating number:";
  cin>>f2;

  cout<<"\nBefore Swapping\n";
  cout<<"\nNum1:"<<num1<<"\nNum2:"<<num2;
  cout<<"\nChar1:"<<ch1<<"\nChar2:"<<ch2;
  cout<<"\nFloat1:"<<f1<<"\nFloat2:"<<f2;


  swap(num1,num2,ch1,ch2,f1,f2);
  cout<<"\n\nAfter swapping";
  cout<<"\nNum1:"<<num1<<"\nNum2:"<<num2;
  cout<<"\nChar1:"<<ch1<<"\nChar2:"<<ch2;
  cout<<"\nFloat1:"<<f1<<"\nFloat2:"<<f2;
}

void swap(int &num1,int &num2,char &ch1,char &ch2,float &f1,float &f2)
{
  int temp1;
  char temp2;
  float temp3;

  temp1=num1;
  num1=num2;
  num2=temp1;

  temp2=ch1;
  ch1=ch2;
  ch2=temp2;

  temp3=f1;
  f1=f2;
  f2=temp3;
}
