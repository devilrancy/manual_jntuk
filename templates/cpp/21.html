#include<iostream.h>

class sam
{
  private:
    int num1,num2;
  public:
    sam(int i,int j)
    {
      num1=i;
      num2=j;
      cout<<"Default constructor working";
    }
    sam(sam &obj)
    {
      
      num1=obj.num2;
      num2=obj.num2;
      cout<<"Copy Constructor working";
    }

    void show()
    {
      cout<<"\nNum1:"<<num1<<"\nNum2:"<<num2;
    }
};

void main()
{
  cout<<"\nCalling Default Constructor.....";
  sam ob1(4,5);
  ob1.show();

  cout<<"\n\nCalling Copy Constructor.....";
  sam ob2(ob1);
  ob2.show();

  cout<<"\n\nAnother way to call Copy Constructor.....";
  sam ob3=ob1;
  ob3.show();
}
