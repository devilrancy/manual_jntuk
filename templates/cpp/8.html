#include<iostream.h>

int factorial(int);

void main()
{
int fact;
  cout<<"Whoes Factorial you want to find:";
  cin>>fact;

  fact=factorial(fact);

  cout<<fact;
}

int factorial(int fact)
{
  if(fact!=1)
    fact*=factorial(fact-1);
  return fact;
}
