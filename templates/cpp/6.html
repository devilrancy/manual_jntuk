#include<iostream.h>
#include<conio.h>
#include<math.h>
#include<stdlib.h>

void main()
{
  float a,b,c,x,x1,x2,rpart,ipart;
  int delta;

  cout<<"\nEnter three numbers a,b &c of ax^2 + bx + c:\n";
  cin>>a>>b>>c;

  if(!a)
    cout<<"\nValue of a should not be zero\nAborting!!!!!";
  else
  {
  delta=b*b-4*a*c;

  if(delta==0)
  {
        cout<<"\n Roots are REAL and EQUAL:\n ";
        x=-b/(2*a);
        cout<<"Root1 and Root2:"<<x;
  }
  else if(delta>0)
  {
         cout<<"\n REAL & DISTINCT roots are:\n ";
         x1=(-b+sqrt(delta))/(2*a);
         x2=(-b-sqrt(delta))/(2*a);
         cout<<"Root1:"<<x1<<"\nRoot2:"<<x2;
  }
  else
        cout<<"\nRoots are COMPLEX AND IMAGINARY";
  }

}
