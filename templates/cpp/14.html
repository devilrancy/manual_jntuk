#include<iostream.h>
void main()
{
  int Numerator,Denominator ;

  cout<<"\nEnter the Numerator:";
  cin>>Numerator;

  cout<<"\nEnter the Denominator:";
  cin>>Denominator;

  try
  {
    if(Denominator!=0)
    {
      float div=(float)Numerator/Denominator;
      cout<<"Result is:"<<div;
    }
    else
    {
      throw(Denominator);
    }
  }

  catch(int d)
  {
    cout<<"\nDenominator cannot be zero";
  }
}
