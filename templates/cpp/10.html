#include <iostream.h>

int add(int, int);
float add(float, float);

int main()
{
  int a, b, x;
  float c, d, y;

  cout << "Enter two integers\n";
  cin >> a >> b;

  x = add(a, b);

  cout << "Sum of two integers is: " << x << endl;

  cout << "Enter two floating point numbers\n";
  cin >> c >> d;

  y = add(c, d);

  cout << "Sum of two floating point numbers is: " << y << endl;

  return 0;
}

int add(int x, int y)
{
  return (x+y);
}

float add(float x, float y)
{
  return (x+y);
}
