#include<iostream.h>

class Inc{
  int val;
public:
  Inc() {
    val=0;
  }

  Inc(int v)  {
    val=v;
  }

  Inc operator++()  {
    ++val;
    return Inc(val);
  }

  Inc operator++(int) {
    Inc I(val);
    ++val;
    return I;
  }

  void show() {
    cout<<"Value:"<<val;
  }
};

void main()
{
  Inc obj(15);
  cout<<"Before ";
  obj.show();
  ++obj;
  cout<<"\nAfter ";
  obj.show();
}
