#include<iostream.h>

void callbyvalue(int);
void callbyreference(int &);
void main()
{
  int num;

  cout<<"\nEnter 1st number:";
  cin>>num;

  cout<<"\nCall by Value\n";
  cout<<"\nValue of Number before calling function:"<<num;
  callbyvalue(num);
  cout<<"\nValue of Number in main:"<<num;

  cout<<"\n\nCall by Reference\n";
  cout<<"\nValue of Number before calling function:"<<num;
  callbyreference(num);
  cout<<"\nValue of Number in main:"<<num;
}

void callbyvalue(int num)
{
  cout<<"\nEnter the number with which you want"<<num<<" to swap: ";
  cin>>num;
  cout<<"\nValue of Number in function Call by Value:"<<num;
}

void callbyreference(int &num)
{
  cout<<"\nEnter the number with which you want"<<num<<" to swap: ";
  cin>>num;
  cout<<"\nValue of Number in function Call by Reference:"<<num;
}
