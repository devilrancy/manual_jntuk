#include<iostream.h>

int swap(int &,int &);
int swap(char &,char &);
int swap(float &,float &);

void main()
{
  int num1,num2;
  float f1,f2;
  char ch1,ch2;

  cout<<"\nEnter the 1st Number:";
  cin>>num1;

  cout<<"\nEnter the 2nd Number:";
  cin>>num2;

  cout<<"\nEnter the 1st Character:";
  cin>>ch1;

  cout<<"\nEnter the 2nd Character:";
  cin>>ch2;

  cout<<"\nEnter the 1st Floating Number:";
  cin>>f1;

  cout<<"\nEnter the 2nd Floating Number:";
  cin>>f2;

  cout<<"\nBefore Swapping\n";
  cout<<"\n1st Number:"<<num1<<"\n2nd Number:"<<num2;
  swap(num1,num2);
  cout<<"\n\nAfter Swapping";
  cout<<"\n1st Number:"<<num1<<"\n2nd Number:"<<num2;

  cout<<"\n\nBefore Swapping\n";
  cout<<"\n1st Character:"<<ch1<<"\n2nd Character:"<<ch2;
  swap(ch1,ch2);
  cout<<"\n\nAfter Swapping";
  cout<<"\n1st Character:"<<ch1<<"\n2nd Character"<<ch2;

  cout<<"\n\nBefore Swapping\n";
  cout<<"1st Floating Number:"<<f1<<"\n2nd Floating Number:"<<f2;
  swap(f1,f2);
  cout<<"\n\nAfter Swapping";
  cout<<"\n1st Floating Number:"<<f1<<"\n2nd Floating Number:"<<f2;

}

int swap(int &num1,int &num2)
{
int temp;
  temp=num1;
  num1=num2;
  num2=temp;
  return 0;
}

int swap(char &ch1,char &ch2)
{
char temp;
  temp=ch1;
  ch1=ch2;
  ch2=temp;
  return 0;
}

int swap(float &f1,float &f2)
{
float temp;
  temp=f1;
  f1=f2;
  f2=temp;
  return 0;
}
