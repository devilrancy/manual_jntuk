#include<iostream.h>
#include<conio.h>
void main(){
  int n,a[20],i,j,temp;
  cout<<"\nEnter the Limit:";
  cin>>n;
  for(i=1;i<=n;i++)  {
    cin>>a[i];
  }
  for(i=1;i<=n;i++)
    for(j=i+1;j<=n;j++)
      if(a[i]>a[j])
      {
        temp=a[i];
        a[i]=a[j];
        a[j]=temp;
      }
  for(i=1;i<=n;i++)
    cout<<"\t"<<a[i];

}
