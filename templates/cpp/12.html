#include<iostream.h>

inline void integer(int num)
{
  (num%2==0)?cout<<"Even Number":cout<<"Odd Number";
}

void main()
{
  int num;
  cout<<"\nEnter the integer:";
  cin>>num;

  integer(num);
}
