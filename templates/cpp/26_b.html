//multiple inheritance
#include<iostream.h>

class vehical
{
public:
  void display()
  {
  cout<<"Its vehical class";
  }
};

class car
{
  public:
    void dis()
    {
      cout<<"\nits a car class";
    }
};

class bike:public car,public vehical
{
  public:
    void displays()
    {
      cout<<"\nIts Bike Class";
    }
};

void main()
{
  bike obj;
  obj.display();
  obj.displays();
  obj.dis();
}
